```js
function add(num) {
  var sum = 0;
  sum = sum + num;
  var tempFun = function (numB) {
    if (arguments.length === 0) {
      return sum;
    } else {
      sum = sum + numB;
      return tempFun;
    }
  };

  tempFun.valueOf = function () {
    return sum;
  };
  tempFun.toString = function () {
    return sum + "";
  };

  return tempFun;
}
```
